def archive_rotowire_json(
    *,
    repo_root: Path,
    rotowire_json_path: Path,
    ids: ArchiveIds,
    dst_name: str = "rotowire_lines.json",
) -> dict:
    """
    Archives the live-used Rotowire lines json into the same IAEL snapshot folder:
      data/archives/iael/YYYY/YYYY-MM-DD/<snapshot_id>/rotowire_lines.json

    This is intentionally additive only:
    - does not change the live write location
    - does not change schema
    - does not affect upstream/downstream model behavior
    - exists only so replay/backtest can recover the exact live-used lines context later
    """
    year = ids.date_dashed[0:4]
    base = repo_root / "data" / "archives"

    iael_snap_dir = base / "iael" / year / ids.date_dashed / ids.snapshot_id
    iael_snap_dir.mkdir(parents=True, exist_ok=True)

    dst = iael_snap_dir / dst_name
    res = {
        "run_id": ids.run_id,
        "snapshot_id": ids.snapshot_id,
        "date": ids.date_dashed,
        "rotowire_snapshot_dir": str(iael_snap_dir),
        "rotowire_src": str(rotowire_json_path),
        "rotowire_dst": str(dst),
        "errors": [],
    }

    try:
        if rotowire_json_path.exists():
            shutil.copy2(rotowire_json_path, dst)
        else:
            res["errors"].append(f"missing: {rotowire_json_path}")
    except Exception as e:
        res["errors"].append(f"copy_failed: {e!r}")

    try:
        (iael_snap_dir / "rotowire_manifest.json").write_text(json.dumps(res, indent=2), encoding="utf-8")
    except Exception as e:
        res["errors"].append(f"manifest_failed: {e!r}")

    return res
