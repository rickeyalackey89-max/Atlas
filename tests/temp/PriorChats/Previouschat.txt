new injury snapshot is pulled/parsed

newest normalized file is republished into normalized/latest.json

Freshness OK: age=0.0 min

Phase B — Role context was stuck in “no_outs” because injury normalized rows had no budget

We inspected normalized/latest.json and found rows had:

team, player, status, hard_invalid, etc.

but NO out_frac (no “removed budget” signal).

Role context needs a budget signal to compute removed-share allocation; without it, it often collapses to no_outs.

✅ Fix: add deterministic out_frac to injury normalized output

Updated scripts/dev/adhoc/injury/injury_pull_and_parse.py to emit out_frac using a conservative status→budget mapping (e.g., OUT/DOUBTFUL/QUESTIONABLE mapped to fixed fractions).

Verified later that latest.json had:

has_out_frac == rows

and many out_frac_gt0

We found multiple issues that made Oracle and audits misleading:

C1) role_ctx_components was NaN for a huge fraction of legs

In runs, hundreds of rows had role_ctx_components NaN.

This made “no_matches” and diagnostics unreadable.

C2) ok_combo rows had outs_used == 0 and role_ctx_outs == NaN

Oracle reported outs_used_gt0 = 0 everywhere even with many “ok_combo” rows.

We confirmed:

ok_combo rows were large

role_ctx_outs_used unique [0]

We traced this to the combo branch of role context:

ok_combo was being set unconditionally for combos

but the combo role_debug dict did NOT include:

outs, outs_used, by_out, bump

So exporter couldn’t populate role_ctx_outs or role_ctx_outs_used.

Serialize structured role_debug fields (outs, by_out, components, component_mults, component_reasons) to JSON strings when exporting so CSVs never get NaN/object leakage.

import re
_leg_player_re = re.compile(r"^(.*?)\s+(OVER|UNDER)\s+", re.IGNORECASE)

def _leg_player_name(leg: str) -> str:
    s = str(leg or "").strip()
    m = _leg_player_re.match(s)
    if not m:
        return ""
    return m.group(1).strip()

def annotate_q_slips(df: pd.DataFrame, max_legs: int) -> pd.DataFrame:
    if df is None or len(df) == 0:
        return df
    out = df.copy()

    leg_cols = [f"leg_{i}" for i in range(1, max_legs + 1) if f"leg_{i}" in out.columns]
    if not leg_cols:
        # fallback: try "legs" if it exists (pipe-delimited)
        if "legs" in out.columns:
            leg_cols = ["legs"]
        else:
            out["q_leg_count"] = 0
            out["q_players"] = ""
            return out

    q_counts = []
    q_players_str = []

    for _, r in out.iterrows():
        legs = []
        if leg_cols == ["legs"]:
            s = str(r.get("legs", "") or "")
            legs = [p.strip() for p in s.split(" | ") if p.strip()]
        else:
            legs = [str(r.get(c, "") or "").strip() for c in leg_cols if str(r.get(c, "") or "").strip()]

        q_hit = []
        for leg in legs:
            p = _leg_player_name(leg)
            if not p:
                continue
            key = normalize_person_name(p).lower()
            if key in q_set:
                q_hit.append(q_disp.get(key, p))

        # unique, stable order
        seen = set()
        uniq = [x for x in q_hit if not (x in seen or seen.add(x))]

        q_counts.append(len(uniq))
        q_players_str.append(" | ".join(uniq))

    out["q_leg_count"] = q_counts
    out["q_players"] = q_players_str
    return out

    Text-based state-machine parser for modern NBA injury PDFs.

    Emits rows with:
      team (3-letter abbr), player ("Last,First"), status, reason, game_date ("MM/DD/YYYY")
    """

        if not t:
            return ""
        u = t.upper()
        if len(u) == 3 and u.isalpha():
            return u
        # CamelCase / long names only if they map to a real NBA abbrev
        return _team_to_abbr(t) or ""

        t = (tok or "").strip()

        if not active:
            return

        if active.get("team") and active.get("player") and active.get("status"):
            out_rows.append(active)

            page_text = page.extract_text() or ""

                    line = piece.strip()
                    if not line:
                        continue

                    if line.startswith("Injury Report:"):
                        continue
                    if line.startswith("Page") and "of" in line:
                        continue
                    if line.startswith("GameDate") and "PlayerName" in line:
                        continue

                    tokens = line.split()
                    if not tokens:
                        continue

                    # and we have an active row, treat it as reason continuation.
                    if active and ("," not in tokens[0]) and (not DATE_PREFIX_RE.match(tokens[0])) and (not any(_is_matchup(t) for t in tokens)):

def parse_txt_rows_text(txt_path: Path) -> List[Dict[str, str]]:
    out_rows: List[Dict[str, str]] = []
    ctx: Dict[str, str] = {"game_date": "", "team": ""}
    active: Optional[Dict[str, str]] = None

    def flush() -> None:
        nonlocal active
        if not active:
            return
        active["reason"] = _clean_ws(active.get("reason", ""))
        if active.get("team") and active.get("player") and active.get("status"):
            out_rows.append(active)
        active = None

    def is_matchup(tok: str) -> bool:
        t = (tok or "").strip()
        return ("@" in t) and (len(t) >= 5) and (len(t) <= 9)

    # treat either "DAL" or "Dallas Mavericks" style via your existing _team_to_abbr
    def team_abbr_from_tokens(tokens: List[str]) -> str:
        s = " ".join(tokens).strip()
        if not s:
            return ""
        u = s.upper()
        if len(u) == 3 and u.isalpha():
            return u
        return _team_to_abbr(s) or ""

    lines = txt_path.read_text(encoding="utf-8", errors="ignore").splitlines()

    for raw in lines:
        line = raw.strip()
        if not line:
            continue

        # Skip header/footer noise
        if line.startswith("Injury Report:"):
            continue
        if line.startswith("Page") and "of" in line:
            continue
        if line.startswith("Game Date") or line.startswith("GameDate"):
            continue

        # If the line is a pure NOT YET SUBMITTED placeholder row, skip
        if "NOT YET SUBMITTED" in line.upper():
            continue

        tokens = line.split()
        if not tokens:
            continue

        # REASON continuation: no date, no matchup, no comma-leader, and we have active
        if active and (not DATE_PREFIX_RE.match(tokens[0])) and (not any(is_matchup(t) for t in tokens)) and ("," not in line):
            active["reason"] = (active.get("reason", "") + " " + line).strip()
            continue

        # DATE row: starts with MM/DD/YYYY
        if DATE_PREFIX_RE.match(tokens[0]):
            # Find matchup token
            mi = None
            for i, tok in enumerate(tokens):
                if is_matchup(tok):
                    mi = i
                    break
            if mi is None or mi + 1 >= len(tokens):
                continue

            ctx["game_date"] = tokens[0]

            # After matchup token, capture team name tokens until player begins.
            rest = tokens[mi + 1 :]
            # Build team tokens from the front of rest:
            team_tokens: List[str] = []
            j = 0
            while j < len(rest):
                team_tokens.append(rest[j])
                # Stop when next token (or current) suggests player begins (comma in current or next)
                if ("," in rest[j]) or (j + 1 < len(rest) and "," in rest[j + 1]):
                    # we accidentally included part of player; remove and stop
                    team_tokens.pop()
                    break
                # team names in report are typically 2-4 tokens; cap to avoid runaway
                if len(team_tokens) >= 4:
                    break
                j += 1

            team_u = team_abbr_from_tokens(team_tokens)
            if not team_u:
                continue
            ctx["team"] = team_u

            player_tokens = rest[len(team_tokens) :]
            psr = _parse_line_player_status_reason(player_tokens)
            if psr:
                player, status, reason = psr
                flush()
                active = {"team": ctx["team"], "player": player, "status": normalize_status(status), "reason": reason, "game_date": ctx["game_date"]}
            continue

        # TEAM HEADER row: starts with a spaced team name then player
        # Example: "Orlando Magic Black, Anthony Questionable ..."
        # We attempt prefixes up to 4 tokens for team name and require that remaining has a comma player.
        found_header = False
        for n in (4, 3, 2, 1):
            if len(tokens) <= n:
                continue
            team_u = team_abbr_from_tokens(tokens[:n])
            if team_u and ("," in " ".join(tokens[n:])):  # remaining includes a player format
                ctx["team"] = team_u
                psr = _parse_line_player_status_reason(tokens[n:])
                if psr:
                    player, status, reason = psr
                    flush()
                    active = {"team": ctx["team"], "player": player, "status": normalize_status(status), "reason": reason, "game_date": ctx["game_date"]}
                found_header = True
                break
        if found_header:
            continue

        # PLAYER row: begins with Last,First (continuation under current team)
        if "," in tokens[0]:
            if not ctx["team"]:
                continue
            psr = _parse_line_player_status_reason(tokens)
            if psr:
                player, status, reason = psr
                flush()
                active = {"team": ctx["team"], "player": player, "status": normalize_status(status), "reason": reason, "game_date": ctx["game_date"]}
            continue

        # Otherwise ignore
        continue

    flush()

    cleaned: List[Dict[str, str]] = []
    for r in out_rows:
        if r.get("player", "").strip() in {"PlayerName", "PLAYERNAME"}:
            continue
        if r.get("status", "").strip() in {"CURRENTSTATUS", "STATUS"}:
            continue

    if not cleaned:
        print(f"[INJURY] No injury rows parsed from TXT: {txt_path}")
