from __future__ import annotations

import sys
from datetime import datetime
from pathlib import Path

import pandas as pd

PROJECT_ROOT = Path(__file__).resolve().parents[1]
ROSTER_MAP_PATH = PROJECT_ROOT / "data" / "input" / "roster_map.csv"
AUDIT_DIR = PROJECT_ROOT / ".atlas_audit" / "diagnostics"

VALID_TEAMS = {
    "ATL", "BOS", "BKN", "CHA", "CHI", "CLE", "DAL", "DEN", "DET",
    "GSW", "HOU", "IND", "LAC", "LAL", "MEM", "MIA", "MIL", "MIN",
    "NOP", "NYK", "OKC", "ORL", "PHI", "PHX", "POR", "SAC", "SAS",
    "TOR", "UTA", "WAS",
}


def _season_string_for_today(today: datetime | None = None) -> str:
    today = today or datetime.now()
    year = today.year
    # NBA season rolls over in October.
    start_year = year if today.month >= 10 else year - 1
    end_year_short = str((start_year + 1) % 100).zfill(2)
    return f"{start_year}-{end_year_short}"


def _is_valid_team(team: str) -> bool:
    return isinstance(team, str) and team in VALID_TEAMS


def _load_old_map() -> dict[str, str]:
    if not ROSTER_MAP_PATH.exists():
        return {}
    df = pd.read_csv(ROSTER_MAP_PATH)
    if "player" not in df.columns or "team" not in df.columns:
        raise RuntimeError(f"Unexpected roster_map schema in {ROSTER_MAP_PATH}")
    df["player"] = df["player"].astype(str).str.strip()
    df["team"] = df["team"].astype(str).str.strip().str.upper()
    df = df[(df["player"] != "") & (df["team"].apply(_is_valid_team))]
    return dict(zip(df["player"], df["team"]))


def _fetch_full_league_roster() -> dict[str, str]:
    try:
        from nba_api.stats.endpoints import commonallplayers
    except Exception as exc:
        raise RuntimeError(
            "nba_api.stats.endpoints.commonallplayers is required for roster refresh"
        ) from exc

    seasons_to_try = [
        _season_string_for_today(),
        _season_string_for_today(datetime(datetime.now().year - 1, 11, 1)),
    ]

    last_exc = None
    for season in seasons_to_try:
        try:
            endpoint = commonallplayers.CommonAllPlayers(
                is_only_current_season=1,
                league_id="00",
                season=season,
            )
            df = endpoint.get_data_frames()[0].copy()
            expected = {"DISPLAY_FIRST_LAST", "TEAM_ABBREVIATION"}
            if not expected.issubset(df.columns):
                raise RuntimeError(
                    f"CommonAllPlayers missing expected columns. Got: {list(df.columns)}"
                )

            df["player"] = df["DISPLAY_FIRST_LAST"].astype(str).str.strip()
            df["team"] = df["TEAM_ABBREVIATION"].astype(str).str.strip().str.upper()

            df = df[(df["player"] != "") & (df["team"].apply(_is_valid_team))]
            df = df.drop_duplicates(subset=["player"], keep="first").sort_values("player")

            roster_map = dict(zip(df["player"], df["team"]))
            if roster_map:
                print(f"[ROSTER_MAP] Loaded {len(roster_map)} players from NBA API for season {season}")
                return roster_map
        except Exception as exc:
            last_exc = exc

    raise RuntimeError(f"Failed to fetch full NBA roster from nba_api: {last_exc}")


def _write_outputs(old_map: dict[str, str], new_map: dict[str, str]) -> None:
    ROSTER_MAP_PATH.parent.mkdir(parents=True, exist_ok=True)
    AUDIT_DIR.mkdir(parents=True, exist_ok=True)

    roster_df = pd.DataFrame(
        [{"player": player, "team": team} for player, team in sorted(new_map.items())]
    )
    roster_df.to_csv(ROSTER_MAP_PATH, index=False)

    updates = []
    all_players = sorted(set(old_map) | set(new_map))
    for player in all_players:
        old_team = old_map.get(player, "")
        new_team = new_map.get(player, "")
        if old_team != new_team:
            updates.append(
                {
                    "player": player,
                    "old_team": old_team,
                    "new_team": new_team,
                    "source": "nba_current_roster",
                }
            )

    ts = datetime.now().strftime("%Y%m%d_%H%M%S")
    audit_path = AUDIT_DIR / f"roster_map_updates_{ts}.csv"
    updates_df = pd.DataFrame(updates, columns=["player", "old_team", "new_team", "source"])
    updates_df.to_csv(audit_path, index=False)

    print(f"[ROSTER_MAP] Wrote {len(new_map)} entries to {ROSTER_MAP_PATH}")
    print(f"[ROSTER_MAP] Wrote {len(updates)} updates to {audit_path}")


def main() -> int:
    old_map = _load_old_map()
    new_map = _fetch_full_league_roster()
    _write_outputs(old_map, new_map)
    return 0


if __name__ == "__main__":
    try:
        raise SystemExit(main())
    except Exception as exc:
        print(f"[ROSTER_MAP][ERROR] {exc}", file=sys.stderr)
        raise

def _coerce_iael_frame(obj: Any) -> pd.DataFrame:
    if isinstance(obj, list):
        df = pd.DataFrame(obj)
    elif isinstance(obj, dict):
        if isinstance(obj.get("rows"), list):
            df = pd.DataFrame(obj["rows"])
        elif isinstance(obj.get("data"), list):
            df = pd.DataFrame(obj["data"])
        else:
            df = pd.DataFrame([obj])
    else:
        df = pd.DataFrame()

    if df.empty:
        return df

    # Normalize likely key columns used downstream
    if "player" not in df.columns:
        for c in ("player_name", "name", "PLAYER"):
            if c in df.columns:
                df["player"] = df[c]
                break
    if "player" in df.columns:
        df["player"] = df["player"].astype(str).str.strip()
        df["player_norm"] = _normalize_player(df["player"])

    if "team" not in df.columns:
        for c in ("team_u", "abbr", "TEAM", "team_abbr"):
            if c in df.columns:
                df["team"] = df[c]
                break
    if "team" in df.columns:
        df["team"] = df["team"].astype(str).str.upper().str.strip()
        df["team_u"] = df["team"]

    if "status" in df.columns:
        df["status"] = df["status"].astype(str).str.upper().str.strip()

    if "report_date" in df.columns:
        df["report_date"] = pd.to_datetime(df["report_date"], errors="coerce").dt.normalize()

    return df

def _pick_historical_iael_file(iael_dir: Path, snap: Path, board_df: pd.DataFrame) -> Path | None:
    if not iael_dir.exists():
        return None

    snap_dt = _parse_date_from_filename(snap)
    if snap_dt is None and "game_date" in board_df.columns:
        vals = pd.to_datetime(board_df["game_date"], errors="coerce").dropna()
        if not vals.empty:
            snap_dt = vals.iloc[0].normalize()

    if snap_dt is None:
        return None

    day_str = snap_dt.strftime("%Y-%m-%d")

    # Prefer exact dated normalized snapshots like 2026-03-06_09_15PM.json
    cands = sorted(iael_dir.glob(f"{day_str}_*.json"))
    if cands:
        return cands[-1]

    # Fallback: exact day file without suffix
    p = iael_dir / f"{day_str}.json"
    if p.exists():
        return p

    return None


def load_historical_iael_for_snapshot(iael_dir: Path, snap: Path, board_df: pd.DataFrame) -> pd.DataFrame | None:
